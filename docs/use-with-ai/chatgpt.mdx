ChatGPT can use tax-engine through function calling. You define functions that wrap the `tax` binary, and ChatGPT calls them like any other tool.

## Python wrapper

```python
import subprocess
import json

TAX_BIN = "/usr/local/bin/tax"  # path to the downloaded binary

def call_tax(args: list[str]) -> dict:
    result = subprocess.run([TAX_BIN] + args, capture_output=True, text=True)
    if result.returncode != 0:
        return {"error": result.stderr}
    return json.loads(result.stdout)
```

## Dispatch function calls

```python
def dispatch_tool(name: str, args: dict) -> str:
    if name == "tax_return_create":
        result = call_tax(["return", "create", "--year", str(args["year"])])
    elif name == "tax_form_add":
        result = call_tax([
            "form", "add",
            "--returnId", args["returnId"],
            "--node_type", args["node_type"],
            json.dumps(args["data"])
        ])
    elif name == "tax_return_get":
        result = call_tax(["return", "get", "--returnId", args["returnId"]])
    elif name == "tax_return_validate":
        result = call_tax(["return", "validate", "--returnId", args["returnId"]])
    elif name == "tax_node_inspect":
        result = call_tax(["node", "inspect", "--node_type", args["node_type"], "--json"])
    else:
        result = {"error": f"Unknown tool: {name}"}
    
    return json.dumps(result)
```

## Complete agent loop

```python
from openai import OpenAI

client = OpenAI()

def run_tax_agent(user_message: str):
    messages = [
        {
            "role": "system",
            "content": (
                "You are a tax preparation assistant. "
                "Use the tax tools to prepare accurate IRS Form 1040 returns. "
                "Always create a return first, add all documents, then compute results."
            )
        },
        {"role": "user", "content": user_message}
    ]
    
    while True:
        response = client.chat.completions.create(
            model="gpt-4o",
            messages=messages,
            tools=tools
        )
        
        message = response.choices[0].message
        messages.append(message)
        
        if not message.tool_calls:
            return message.content
        
        for tool_call in message.tool_calls:
            result = dispatch_tool(
                tool_call.function.name,
                json.loads(tool_call.function.arguments)
            )
            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": result
            })

# Use it
answer = run_tax_agent(
    "I'm single. My W-2 shows $85,000 wages and $12,000 withheld. "
    "I also have $420 in bank interest. Prepare my 2025 federal return."
)
print(answer)
```
