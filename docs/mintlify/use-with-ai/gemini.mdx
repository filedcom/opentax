Gemini supports function calling with the same subprocess pattern. Download the binary, define the functions, and Gemini calls the CLI automatically.

## Python wrapper

```python
import subprocess
import json

TAX_BIN = "/usr/local/bin/tax"

def run_tax(args: list) -> str:
    result = subprocess.run([TAX_BIN] + args, capture_output=True, text=True)
    return result.stdout if result.returncode == 0 else result.stderr
```

## Function definitions

```python
import google.generativeai as genai
from google.generativeai.types import FunctionDeclaration, Tool

tax_tools = Tool(function_declarations=[
    FunctionDeclaration(
        name="tax_return_create",
        description="Create a new tax return. Returns a returnId.",
        parameters={
            "type": "object",
            "properties": {
                "year": {"type": "integer", "description": "Tax year, e.g. 2025"}
            },
            "required": ["year"]
        }
    ),
    FunctionDeclaration(
        name="tax_form_add",
        description="Add a tax document to an existing return.",
        parameters={
            "type": "object",
            "properties": {
                "returnId": {"type": "string"},
                "node_type": {"type": "string", "description": "Form type: w2, f1099int, schedule_c, general, etc."},
                "data": {"type": "object", "description": "Form field values as JSON"}
            },
            "required": ["returnId", "node_type", "data"]
        }
    ),
    FunctionDeclaration(
        name="tax_return_get",
        description="Compute and return all calculated values for a return.",
        parameters={
            "type": "object",
            "properties": {
                "returnId": {"type": "string"}
            },
            "required": ["returnId"]
        }
    ),
    FunctionDeclaration(
        name="tax_return_validate",
        description="Validate a return against IRS MeF business rules.",
        parameters={
            "type": "object",
            "properties": {
                "returnId": {"type": "string"}
            },
            "required": ["returnId"]
        }
    ),
])
```

## Dispatch and agent loop

```python
def dispatch(name: str, args: dict) -> str:
    if name == "tax_return_create":
        return run_tax(["return", "create", "--year", str(args["year"])])
    elif name == "tax_form_add":
        return run_tax([
            "form", "add",
            "--returnId", args["returnId"],
            "--node_type", args["node_type"],
            json.dumps(args["data"])
        ])
    elif name == "tax_return_get":
        return run_tax(["return", "get", "--returnId", args["returnId"]])
    elif name == "tax_return_validate":
        return run_tax(["return", "validate", "--returnId", args["returnId"]])
    return json.dumps({"error": f"Unknown function: {name}"})


def run_tax_agent(user_message: str) -> str:
    genai.configure(api_key="YOUR_API_KEY")
    model = genai.GenerativeModel(
        model_name="gemini-1.5-pro",
        tools=[tax_tools],
        system_instruction=(
            "You are a tax preparation assistant. "
            "Use the tax tools to prepare accurate IRS Form 1040 returns. "
            "Always create a return first, then add documents, then compute."
        )
    )
    
    chat = model.start_chat()
    response = chat.send_message(user_message)
    
    while response.candidates[0].content.parts[0].function_call.name:
        fc = response.candidates[0].content.parts[0].function_call
        result = dispatch(fc.name, dict(fc.args))
        response = chat.send_message(
            genai.protos.Part(
                function_response=genai.protos.FunctionResponse(
                    name=fc.name,
                    response={"result": result}
                )
            )
        )
    
    return response.text


# Use it
print(run_tax_agent(
    "I'm single. W-2: $78,000 wages, $11,000 withheld. "
    "Compute my 2025 federal tax return."
))
```

## Gemini in the browser

You can also use Gemini's built-in code execution in Gemini.google.com. Paste a prompt like:

```
Download the tax engine binary from:
https://github.com/filed-app/tax-engine/releases/latest/download/tax-linux-x64

Make it executable, then compute a 2025 federal tax return:
- Filing status: single  
- W-2: $78,000 wages, $11,000 withheld

Show me the refund or amount owed.
```

Gemini's code execution sandbox can fetch and run the binary directly.
